### Data manipulation

``` r
homicide_df = 
  homicide_raw %>% 
  janitor::clean_names() %>% 
  mutate(city_state = str_c(city, state, sep = ", "))

##total number of homicides by cities
df1 = homicide_df %>% 
  group_by(city_state) %>% 
  summarize(n_total = n()) 
  
##number of unsolved homicides by cities
df2 = homicide_df %>% 
  filter(disposition != "Closed by arrest") %>% 
  group_by(city_state) %>% 
  summarize(n_unsolved = n())
```

``` r
homicide_prop = 
  left_join(df1, df2) %>% 
  filter(city_state == "Baltimore, MD")
```
